\begin{algorithme}
    \begin{enregistrement}{TableDeCodage}
        \champEnregistrement{tableDOctet}{\tableauUneDimension{0..255}{de}{\codebinaire}}
        \champEnregistrement{tableDePresence}{\tableauUneDimension{0..255}{de }{\booleen}}
    \end{enregistrement}
    \\
    \fonction{creerTableCodage}{}{\tabledecodage}
    {}
    {tdc : \tabledecodage}
    {   
        \pour{octet}{ 0}{255}{}{
            \affecter{tdc.tableDePresence[octet]}{\textbf{Faux}}
        }
        \retourner{tdc}
    }
    \\
    \procedure{ajouterCodage}{\paramEntreeSortie{tdc : \tabledecodage}, \paramEntree{o : \octet, cb : \codebinaire}}
    {non(octetPresent(t, octet))}
    {}
    {
        \affecter{octet}{octetVersNaturel(o)}
        \affecter{tdc.tableDOctet[octet]}{cb}
        \affecter{tdc.tableDePresence[octet]}{\textbf{Vrai}}
    }
    \\
    \fonction{octetVersCodeBinaire}{tdc : \tabledecodage, o : \octet}{\codebinaire}
    {octetPresent(t, octet)}
    {}
    {
        \retourner{tdc.tableDOctet[octetVersNaturel(o)]}
    }
    \\
    \fonction{octetPresent}{tdc : \tabledecodage, o : \octet}{\booleen}
    {}
    {}
    {
        \retourner{tdc.tableDePresence[octetVersNaturel(o)]}
    }
\end{algorithme}
