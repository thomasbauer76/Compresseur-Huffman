\begin{algorithme}
    \fonction{lireStatistiques}{fb : \fichierbinaire}{\statistiques}{estOuvert(fb) et (mode(fb) $=$ lecture)}
    {s : \statistiques, octet : \octet, occurrence : \naturel}
    {
    	\affecter{s}{statistiques()}
    	\\
    	\commentaire{La fonction lireStatistiques est utilis√©e quand le curseur est au bon endroit et l'on peut donc directement lire les statistiques qui sont des naturels}
    	
    	\repeter
    	{
    		\instruction{lireNaturel(fb, occurrence)}
    		\sialors{(occurrence $!=$ 0)}
    		{
    			\instruction{lireOctet(fb, octet)}
        		\instruction{fixerOccurrence(s, octet, occurrence)}
        	}
    	}{occurrence $=$ 0}
    \retourner{s}
    }
\end{algorithme}
