\begin{algorithme}
    \procedure{decoder}{\paramEntree{fichierBinaireEntrant : \fichierbinaire, arbreDeHuffman : \ArbreDeHuffman}, \paramSortie{fichierBinaireSortant : \fichierbinaire}}{}
    {}
    {
        \sialors{\textbf{non} \textproc{estUneFeuille}(arbreDeHuffman)}
        {
            \textproc{seDeplacerDansLArbre}(arbreDeHuffman, fichierBinaireEntrant, fichierBinaireSortant)
        }
    }
    
    \procedure{seDeplacerDansLArbre}{\paramEntree{arbreCourant : \ArbreDeHuffman, fichierBinaireEntrant : \fichierbinaire}, \paramSortie{fichierBinaireSortant : \fichierbinaire}}{}
    {}
    {
        \sialors{\textbf{non} \textproc{estUneFeuille}(arbreCourant)}
        {
            bitCourant = \textproc{lireBit}(fichierBinaireEntrant)
            
            \sialors{bitCourant = 0}
            {
                \textproc{seDeplacerDansLArbre}(\textproc{obtenirFilsGauche}(arbreCourant), fichierBinaireEntrant, fichierBinaireSortant)
            }
            \sialors{bitCourant = 1}
            {
                \textproc{seDeplacerDansLArbre}(\textproc{obtenirFilsDroit}(arbreCourant), fichierBinaireEntrant, fichierBinaireSortant)
            }
        }
        \sialors{\textproc{estUneFeuille}(arbreCourant)}
        {
            \textproc{ecrireOctet}(\textproc{obtenirOctet}(arbreCourant), fichierBinaireSortant)
            \textproc{seDeplacerDansLArbre}(arbreDeHuffman, fichierBinaireEntrant, fichierBinaireSortant)
        }
    }
\end{algorithme}
