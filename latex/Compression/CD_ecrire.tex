\begin{algorithme}
    \procedure{ecrireIdentifiant}{\paramEntreeSortie{fb : \fichierbinaire}}{}
    {id : Naturel}
    {
    	 \affecter{id}{1000}
    	 \instruction{ecrireNaturel(fb,id)}
        
    }
    \procedure{ecrireStatistiques}{\paramEntreeSortie{fb : \fichierbinaire}, \paramEntree{s : \statistiques}}{}{o: \naturel}
    {
    		\pour{o}{ 0}{255}{}{
			    \instruction{ecrireNaturel(fb, obtenirOccurence(s,o))}
    		}
    }
    
    
    \procedure{ecrireTailleFichier}{\paramEntreeSortie{fb : \fichierbinaire},\paramEntree{taillefb : \naturel}}{}{}
    {
        \instruction{ecrireNaturel(fb, taillefb)}
    }
\end{algorithme}
