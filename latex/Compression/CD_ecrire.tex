\begin{algorithme}
    \procedure{ecrireIdentifiant}{\paramEntreeSortie{fb : \fichierbinaire}}{}
    {}
    {
    	 \instruction{ecrireoctet(fb,creeoctet(bitA0,bitA0,bitA0,bitA0,bitA0,bitA0,bitA1,bitA1))}
         \instruction{ecrireoctet(fb,creeoctet(bitA1,bitA1,bitA1,bitA0,bitA1,bitA0,bitA0,bitA0))}
    }
    \procedure{ecrireStatistique}{\paramEntreeSortie{fb : \fichierbinaire}}{}{i, occ : Naturel, o : \octet, s : \statistiques}
    {
    	
    	\pour{o}{0}{255}{}{
			\instruction{obtenirStatistiques(fb,s)}
			\instruction{ecrireoctet(fb,o)}
			\instruction{ecrireoctet(fb,naturelVersOctet(obtenirOccurence(s,o)))}
    	}
    }
    
    
     \procedure{ecrireTailleFichier}{\paramEntreeSortie{fb : \fichierbinaire},\paramEntree{taillefb : Naturel}}{}{}{
     \instruction{ecrireoctet(naturelVersOctet(taillefb),fb)}
     }
\end{algorithme}
