\begin{algorithme}
    \procedure{concatenerCodeBinaireDansFichier}{\paramEntreeSortie{fbCompresse : \fichierbinaire, cbtemp : \codebinaire},\paramEntree{cb : \codebinaire}}
	{estOuvert(fbCompresse) et (mode(fbCompresse) $=$ Ã©criture)}
    {i, tailleCb: \naturel}
    {
		\affecter{i}{0}
		\affecter{tailleCb}{obtenirLongueur(cb)}	
		\\
		\tantque{i $\neq$ tailleCb}
		{
			\sialors{obtenirLongueur(cbTemp) = 8}
			{											
				\affecter{cbTemp}{creerCodeBinaire(obtenirIemeBit(cb, i))}
				\affecter{i}{i + 1}
			}
			\\
			\tantque{i < tailleCb et obtenirLongueur(cbTemp) < 8}
			{
				\instruction{ajouterBit(cbTemp, obtenirIemeBit(cb, i))}
				\affecter{i}{i + 1}
			}  
			\\
			\sialors{obtenirLongueur(cbTemp) = 8}
			{											
				\instruction{ecrireOctet(fbCompresse, codeBinaireEnOctet(cbtemp))}
			}
		}
    }
\end{algorithme}
