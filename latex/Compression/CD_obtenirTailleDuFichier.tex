\begin{algorithme}
    \fonction{obtenirTailleDuFichier}{fb : \fichierbinaire}{Naturel}{}{taille : Naturel, o : \octet}
    {
        \affecter{taille}{0}
        \instruction{ouvrir(fb,lecture)}
        
       \tantque{$non$ finfichier(fb)}{
        	\instruction{lireoctet(fb,o)}
        	\affecter{taille}{taille $+$ 1}
	    }
	    \retourner{taille}
    }
\end{algorithme}
