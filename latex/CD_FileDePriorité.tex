\begin{algorithme}
    \procedure{enfiler}{\paramEntreeSortie{file : FileDePriorite}, \paramEntree{e : Element}}{}
    {temp : FileDePriorite}
    {
         \sialorssinon{estVide(file) ou e $\leq$ ObtenirElement(file)}
         {
            \affecter{temp}{file}
            \instruction{allouer(file)}
            \affecter{obtenirElement(file)}{e}
            \affecter{obtenirListeSuivante(file)}{temp}
         }
         {
            \instruction{enfiler(ObtenirListeSuivant(file), e)}{}
         }
    }
\end{algorithme}

\begin{algorithme}
    \procedure{Defiler}{\paramSortie{e : Element}, \paramEntreeSortie{file : FileDePriorite}}{non(estVide(file))}
    {temp : FileDePriorite}
    {
        \affecter{e}{ObtenirElement(file)}
        \affecter{temp}{file}
        \affecter{file}{obtenirListeSuivante(temp)}
        \instruction{desallouer(temp)}
    }
\end{algorithme}
