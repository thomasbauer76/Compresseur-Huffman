\subsection{Conception Détaillée : TableDeCodage}

\begin{algorithme}
    \begin{enregistrement}{TableDeCodage}
        \champEnregistrement{tableDOctets}{\tableauUneDimension{0..255}{d'}{\octet}}
        \champEnregistrement{tableDePresences}{\tableauUneDimension{0..255}{de }{\booleen}}
    \end{enregistrement}
    \\
    \fonction{creerTableCodage}{}{\tabledecodage}
    {}
    {tdc : \tabledecodage}
    {   
        \pour{octet}{ 0}{255}{}{
            \affecter{tdc.tableDePresences[octet]}{\textbf{Faux}}
        }
        \retourner{tdc}
    }
    \\
    \procedure{ajouterCodage}{\paramEntreeSortie{tdc : \tabledecodage}, \paramEntree{o : \octet, cb : \codebinaire}}
    {non(octetPresent(t, octet))}
    {}
    {
        \affecter{octet}{octetVersNaturel(o)}
        \affecter{tdc.tableDOctets[octet]}{cb}
        \affecter{tdc.tableDePresences[octet]}{\textbf{Vrai}}
    }
    \\
    \fonction{octetVersCodeBinaire}{tdc : \tabledecodage, o : \octet}{\codebinaire}
    {octetPresent(t, octet)}
    {}
    {
        \retourner{tdc.tableDOctets[octetVersNaturel(o)]}
    }
    \\
    \fonction{octetPresent}{tdc : \tabledecodage, o : \octet}{\booleen}
    {}
    {}
    {
        \retourner{tdc.tableDePresences[octetVersNaturel(o)]}
    }
\end{algorithme}
